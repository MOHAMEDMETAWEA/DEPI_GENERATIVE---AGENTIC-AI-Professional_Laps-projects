-- DROP DATABASE IF EXISTS "company-depi";

CREATE DATABASE "company-depi"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'English_United States.1252'
    LC_CTYPE = 'English_United States.1252'
    LOCALE_PROVIDER = 'libc'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;


create table department(
id int primary key,
name varchar(50),
mgrssn int , 
location varchar(50)
)

create table employee(
id int primary key ,
fname varchar(50),
lname varchar(50),
salary decimal ( 10,2),
deptid int , 
foreign key (deptid) references department(id),
address varchar(100),
supervisor_id int,
hire_date date

)

create table employees_retired(
id int primary key , 
name varchar(100)
)

create table job_grades(
grade char(1),
lowsal decimal (10,2),
highsal decimal (10,2)

)


create table wokrs_on(
emp_id int,
pno int
)

create table project(
pnumber int primary key,
pname varchar(50)
)

create table dept_location(
dnumber int,
dlocation varchar(100)
)
